```javascript

const store = unit({
  state: [
    { title: 'Todo 1' },
    { title: 'Todo 2', completed: true }
  ],
  get completed() {
    return this.state.filter(i => i.completed);
  },
  add(title, completed = false) {
    this.state = [ ...this.state, { title, completed }];
  },
  toggle(i) {
    const { state } = this;
    const index = state.indexOf(i);
    this.state = [
      ...state.slice(0, index),
      { ...i, completed: !i.completed },
      ...state.slice(index+1)
    ];
  }
})

```
